program menghitung_diskon_dan_total_belanja;
var     nilai_belanja:integer;     input_diskon, diskon, harga_belanja:real;
const     ketentuan_diskon = 100000;     skala_persentase = 100;
begin     writeln('Selamat datang di aplikasi Penghitung Diskon Dan Total Belanja');     write('Silahkan tekan enter untuk melanjutkan..');     readln();     writeln();     write('Masukan jumlah nilai belanja pembeli : ');readln(nilai_belanja);     if(nilai_belanja >= ketentuan_diskon)then     begin        write('Masukan jumlah diskon dalam skala persentase 1 s/d 100 : ');readln(input_diskon);        diskon        := nilai_belanja * input_diskon / skala_persentase;        harga_belanja := nilai_belanja - diskon;        writeln();        writeln('Pembeli mendapat diskon sebesar Rp. ',round(diskon));        writeln('Total nilai belanja yang harus dibayar adalah Rp. ',round(harga_belanja));     end;     if(nilai_belanja < ketentuan_diskon)then     begin        writeln();        writeln('Pembeli tidak mendapat diskon');        writeln('Total nilai belanja yang harus dibayar adalah Rp. ',nilai_belanja);     end;     writeln();     write('Silahkan tekan enter untuk keluar dari aplikasi..');     readln();
end.
